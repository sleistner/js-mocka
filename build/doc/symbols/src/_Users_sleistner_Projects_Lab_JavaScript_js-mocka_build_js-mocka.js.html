  3     function mockCollection(collection, binding, callback) {
  4         if (JSMocka.isArray(collection)) {
  5             for (var i = 0, len = collection.length; i < len; i++) {
  6                 callback.call(binding, collection[i]);
  7             }
  8         } else {
  9             for (var name in collection) {
 10                 callback.call(binding, name).returns(collection[name]);
 11             }
 12         }
 13         return binding;
 14     }

 17     function mockify(method) {
 18         return function(object, attributes) {
 19             var mocked = new JSMocka(object || {}, attributes);
 20             if (attributes && typeof attributes !== 'function') {
 21                 mocked[method](attributes);
 22             }
 23             return mocked;
 24         };

 27     /** @inner */
 28     function onStubbing(object, method) {
 29         if (JSMocka.Configuration.shouldPrevent('StubbingNonExistentMethod', object)) {
 30             if (!(method in object)) {
 31                 throw new Error('JSMocka::Expectation.Stubbing non existent method: ' + method);
 32             }
 33         }
 34         if (JSMocka.Configuration.shouldPrevent('StubbingNonPublicMethod', object)) {
 35             if ((/^_/).test(method)) {
 36                 throw new Error('JSMocka::Expectation.Stubbing non public method: ' + method);
 37             }
 38         }
 39     }

136         var expectation;

139             if (expectation.isEvaluable()) {

787                 return obj[method] && obj[method]();
788             }
789         },
790 
791         toString: function() {
792             var string = 'responds with "' + method + '"';
793             if (returnValue !== undefined) {
794                 string += ' returning ' + JSMocka.inspect(returnValue);
795             }
796             return string;
797         }
798     };
799 });

879 /**
880 * Expectation
881 * @constructor
882 * @param object {Object} the object to define an expectation on
883 * @param method {String} the expected method
884 */
885 JSMocka.Expectation = function(object, method) {
886 
887     var original = null,
888         parametersMatcher = new JSMocka.Matcher.ParametersMatcher([new JSMocka.Matcher.AnyParametersMatcher()]),
889         cardinalityMatcher = new JSMocka.Matcher.ExactCardinalityMatcher(1),
890         correctCardinality,
891         correctParameters = true,
892         timesCalled = 0,
893         callArguments = [],
894         returnValues = [],
895         negated = false,
896         evaluable = true,
897         patched = false;
898 
899     /**
900     * @inner
901     * Method that simulates the expected behaviour. It matches the expected
902     * parameters, increments the call counter and returns the expected return
903     * value.
904     */
905     function call() {
906         var args = JSMocka.toArray(arguments);
907         callArguments = callArguments.concat(args);
908         correctParameters = correctParameters && parametersMatcher.match(args);
909         timesCalled++;
910         return returnValues.length > 1 ? returnValues.shift() : returnValues[0];
911     }
912 
913     /**
914     * @inner
915     * Moves the original expected method away and replaces it with a method
916     * showing the expected behaviour.
917     * This method is automatically called.
918     */
919     this.apply = function() {
920         if (!patched) {
921             original = object[method];
922             object[method] = call;
923             patched = true;
924         }
925     };
926 
927     /**
928     * @inner
929     * Restores the original expected method. This method is automatically called
930     * on verify.
931     */
932     this.restore = function() {
933         if (patched) {
934             object[method] = original;
935             original = null;
936         }
937     };
938 
939     /**
940     * @inner
941     * Evaluates if the expectations have been met.
942     */
943     this.evaluate = function() {
944         correctCardinality = cardinalityMatcher.match(timesCalled);
945         var satisfied = correctParameters && correctCardinality;
946         return negated ? !satisfied : satisfied;
947     };
948 
949     /**
950     * @inner
951     * Provides a short description of the expectation.
952     */
953     this.toString = function() {
954         var string = (negated ? "NOT " : "") + "Call of method '" + method;
955         string += "' with " + parametersMatcher.toString();
956         string += ", " + cardinalityMatcher.toString();
957         var returnString = JSMocka.collect(returnValues, function(v) {return JSMocka.inspect(v);}).join(', then ');
958         if (returnString.length > 0) {
959             string += ", returning " + returnString;
960         }
961         return string;
962     };
963 
964     /**
965     * @inner
966     * Provides a detailled description of the expectation, including which
967     * expectations have been satisfied, and which have not.
968     */
969     this.details = function() {
970         var string = 'Parameters were' + (correctParameters ? '' : ' NOT') + ' correctly matched. ';
971         string += 'Expected number of calls was' + (correctCardinality ? '' : ' NOT') + ' matched. ';
972         string += timesCalled + ' times called';
973         if (callArguments.length > 0) {
974             string += ', with ' +
975             JSMocka.collect(callArguments, function(args) {return JSMocka.inspect(args);}).join('; ');
976         }
977         if (negated) {
978             string += ' NOTE! This expectation was negated.';
979         }
980         return string;
981     };
982 
983     /**
984     * Modifies the expectation so that the expected method must be called with
985     * the supplied parameters.
986     * @example
987     * var o = {};
988     * JSMocka.mock(o, 'foo').withArgs('bar');
989     * o.foo('bar); //succeeds
990     * o.foo(); //fails
991     * o.foo(1); //fails
992     *
993     * var o = {};
994     * JSMocka.mock(o, 'foo').withArgs(JSMocka.Matcher.InstanceOf(Foo));
995     *
996     * @param args {string|object|array|number|boolean} Parameter matchers may be supplied as arguments.
997     * @returns {JSMocka.Expectation} the object itself for chaining
998     * @see JSMocka.Matcher
999     */
1000     this.withArgs = function(args) {
1001         parametersMatcher = new JSMocka.Matcher.ParametersMatcher(arguments);
1002         return this;
1003     };
1004 
1005     /**
1006     * Syntactic sugar for complex expectations. Actually does nothing.
1007     * @returns {JSMocka.Expectation} the object itself for chaining
1008     */
1009     this.then = function() {
1010         return this;
1011     };
1012 
1013     /**
1014     * Modifies the expectation so that the expected method must be called the
1015     * specified amount of times.
1016     * @example
1017     * JSMocka.create({}).expects('foo').times(3);
1018     * @param card {number} the expected number of calls
1019     * @returns {JSMocka.Expectation} the object itself for chaining
1020     */
1021     this.times = function(card) {
1022         cardinalityMatcher = new JSMocka.Matcher.ExactCardinalityMatcher(card);
1023         return this;
1024     };
1025 
1026     /**
1027     * Modifies the expectation so that the expected method may be called any
1028     * amount of times.
1029     * @returns {JSMocka.Expectation} the object itself for chaining
1030     */
1031     this.anyTime = function() {
1032         cardinalityMatcher = new JSMocka.Matcher.AnyCardinalityMatcher();
1033         return this;
1034     };
1035 
1036     /**
1037     * Modifies the expectation so that the expected method must not be called at
1038     * all.
1039     * @returns {JSMocka.Expectation} the object itself for chaining
1040     */
1041     this.never = function() {
1042         return this.times(0);
1043     };
1044 
1045     /**
1046     * Modifies the expectation so that the expected method must be called exactly
1047     * once. This is the default setting.
1048     * @returns {JSMocka.Expectation} the object itself for chaining
1049     */
1050     this.once = function() {
1051         return this.times(1);
1052     };
1053 
1054     /**
1055     * Modifies the expectation so that the expected method must be called exactly
1056     * twice.
1057     * @returns {JSMocka.Expectation} the object itself for chaining
1058     */
1059     this.twice = function() {
1060         return this.times(2);
1061     };
1062 
1063     /**
1064     * Modifies the expectation so that the expected method must be called at
1065     * least the specified amount of times.
1066     * @param times {number} the min number number of expected calls
1067     * @returns {JSMocka.Expectation} the object itself for chaining
1068     */
1069     this.atLeast = function(times) {
1070         cardinalityMatcher = new JSMocka.Matcher.MinimumCardinalityMatcher(times);
1071         return this;
1072     };
1073 
1074     /**
1075     * Modifies the expectation so that the expected method must be called at
1076     * most the specified amount of times.
1077     * @param times {number} the max number number of expected calls
1078     * @returns {JSMocka.Expectation} the object itself for chaining
1079     */
1080     this.atMost = function(times) {
1081         cardinalityMatcher = new JSMocka.Matcher.MaximumCardinalityMatcher(times);
1082         return this;
1083     };
1084 
1085     /**
1086     * Modifies the expectation so that the expected method must be called any
1087     * amount of times in the interval specified by from and to.
1088     * @param from {number} the min number number of expected calls
1089     * @param to {number} the max number number of expected calls
1090     * @returns {JSMocka.Expectation} the object itself for chaining
1091     */
1092     this.between = function(from, to) {
1093         cardinalityMatcher = new JSMocka.Matcher.RangeCardinalityMatcher(from, to);
1094         return this;
1095     };
1096 
1097     /**
1098     * Negates the expectation, such that it the evaluation fails if all specified
1099     * criteria are met.
1100     * @example
1101     * // In the following example, no instance of class Foo may be created.
1102     *
1103     * var Foo = new function() {};
1104     * JSMocka.create(Foo).anyInstance().not();
1105     * @returns {JSMocka.Expectation} the object itself for chaining
1106     */
1107     this.not = function() {
1108         negated = true;
1109         return this;
1110     };
1111 
1112     /**
1113     * Modifies the expectation so that when the expected method is called, it
1114     * returns the specified value.
1115     *
1116     * This method may be called with multiple arguments, or multiple times. In
1117     * this case, the values are returned in turn on consecutive calls to the
1118     * method.
1119     * @example
1120     * var o = {};
1121     * m = JSMocka.create(o);
1122     * m.stubs('foo').returns(10, 12);
1123     * m.stubs('bar').returns(14).then().returns(16);
1124     *
1125     * @returns {JSMocka.Expectation} the object itself for chaining
1126     */
1127     this.returns = function() {
1128         returnValues = returnValues.concat(JSMocka.toArray(arguments));
1129         return this;
1130     };
1131 
1132     /**
1133     * @inner
1134     */
1135     this.isEvaluable = function() {
1136         return evaluable;
1137     };
1138 
1139     /**
1140     * @inner
1141     */
1142     this.setEvaluable = function(shouldEvaluate) {
1143         evaluable = shouldEvaluate;
1144         return this;
1145     };
1146 };
